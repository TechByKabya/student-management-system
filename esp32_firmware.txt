// ========================= I2S Setup ===============================
void setupI2S() {
  i2s_config_t cfg = {
      .mode = (i2s_mode_t)(I2S_MODE_MASTER | I2S_MODE_RX),
      .sample_rate = VOICE_SAMPLE_RATE,
      .bits_per_sample = I2S_BITS_PER_SAMPLE_16BIT,
      .channel_format = I2S_CHANNEL_FMT_ONLY_LEFT, // INMP441 mono
      .communication_format =
          (i2s_comm_format_t)(I2S_COMM_FORMAT_I2S | I2S_COMM_FORMAT_I2S_MSB),
      .intr_alloc_flags = ESP_INTR_FLAG_LEVEL1,
      .dma_buf_count = 8,
      .dma_buf_len = 1024,
      .use_apll = false,
      .tx_desc_auto_clear = false,
      .fixed_mclk = 0};
  i2s_pin_config_t pins = {.bck_io_num = I2S_SCK_PIN,
                           .ws_io_num = I2S_WS_PIN,
                           .data_out_num = I2S_PIN_NO_CHANGE,
                           .data_in_num = I2S_SD_PIN};
  i2s_driver_install(I2S_PORT, &cfg, 0, NULL);
  i2s_set_pin(I2S_PORT, &pins);
  i2s_zero_dma_buffer(I2S_PORT);
}

// ========================= WAV Header Builder ======================
void buildWavHeader(uint8_t *h, int dataSize) {
  int fileSize = dataSize + 36;
  int byteRate = VOICE_SAMPLE_RATE * (VOICE_BITS / 8);
  // RIFF header
  h[0] = 'R';
  h[1] = 'I';
  h[2] = 'F';
  h[3] = 'F';
  h[4] = (fileSize) & 0xFF;
  h[5] = (fileSize >> 8) & 0xFF;
  h[6] = (fileSize >> 16) & 0xFF;
  h[7] = (fileSize >> 24) & 0xFF;
  h[8] = 'W';
  h[9] = 'A';
  h[10] = 'V';
  h[11] = 'E';
  // fmt chunk
  h[12] = 'f';
  h[13] = 'm';
  h[14] = 't';
  h[15] = ' ';
  h[16] = 16;
  h[17] = 0;
  h[18] = 0;
  h[19] = 0; // sub-chunk size
  h[20] = 1;
  h[21] = 0; // PCM format
  h[22] = 1;
  h[23] = 0; // mono
  h[24] = (VOICE_SAMPLE_RATE) & 0xFF;
  h[25] = (VOICE_SAMPLE_RATE >> 8) & 0xFF;
  h[26] = (VOICE_SAMPLE_RATE >> 16) & 0xFF;
  h[27] = (VOICE_SAMPLE_RATE >> 24) & 0xFF;
  h[28] = (byteRate) & 0xFF;
  h[29] = (byteRate >> 8) & 0xFF;
  h[30] = (byteRate >> 16) & 0xFF;
  h[31] = (byteRate >> 24) & 0xFF;
  h[32] = (VOICE_BITS / 8);
  h[33] = 0; // block align
  h[34] = VOICE_BITS;
  h[35] = 0; // bits per sample
  // data chunk
  h[36] = 'd';
  h[37] = 'a';
  h[38] = 't';
  h[39] = 'a';
  h[40] = (dataSize) & 0xFF;
  h[41] = (dataSize >> 8) & 0xFF;
  h[42] = (dataSize >> 16) & 0xFF;
  h[43] = (dataSize >> 24) & 0xFF;
}

// ========================= Setup ===================================
void setup() {
  Serial.begin(115200);

  pinMode(BUZZER, OUTPUT);
  digitalWrite(BUZZER, LOW);

  // Semaphores
  displayMutex = xSemaphoreCreateMutex();
  sdMutex = xSemaphoreCreateMutex();
  wifiMutex = xSemaphoreCreateMutex();

  // FreeRTOS EventGroup for voice coordination
  voiceEventGroup = xEventGroupCreate();

  // NOTE: No large RAM buffer allocated — audio goes directly to SD card.
  // This works on ANY ESP32 regardless of whether PSRAM is present.
  Serial.println("[Voice] SD-based recording mode. No RAM buffer needed.");

  // ── Power Rail Stabilization ──────────────────────────────────────
  // Wait for the buck converter 5V → ESP32 LDO 3.3V rail to fully
  // settle before initializing the OLED SSD1306 charge pump.
  // On battery power, the startup current spike (ESP32 boot + I2S mic
  // + INMP441 + PN532 + fingerprint) can droop the 3.3V rail long
  // enough to fail the OLED silently. 1500ms covers the worst case.
  delay(1500);
  Serial.println("[Boot] Power rails settled. Initializing peripherals...");
  // ─────────────────────────────────────────────────────────────────

  Wire.begin(21, 22);

  // OLED — retry up to 5 times in case rail is still settling
  bool oledOk = false;
  for (int attempt = 0; attempt < 5; attempt++) {
    if (display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
      oledOk = true;
      break;
    }
    Serial.printf("[OLED] Init attempt %d failed, retrying...\n", attempt + 1);
    delay(300);
  }
  if (!oledOk) {
    // Last resort: halts with serial message so USB debug still works
    Serial.println(
        F("[OLED] SSD1306 failed after 5 attempts — check 3.3V rail!"));
    for (;;)
      ;
  }
  display.clearDisplay();
  display.setTextColor(SSD1306_WHITE, SSD1306_BLACK);
  display.setTextSize(2);
  display.setCursor(25, 20);
  display.print("TEAM X");
  display.setTextSize(1);
  display.setCursor(15, 45);
  display.print("Initializing...");
  display.display();
  playAudio("startup");
  delay(1500);

  // RTC
  if (!rtc.begin()) {
    rtcReady = false;
  } else {
    rtcReady = true;
    if (rtc.lostPower())
      rtc.adjust(DateTime(F(__DATE__), F(__TIME__)));
  }

  // SD Card (mandatory — halt until present)
  if (xSemaphoreTake(sdMutex, portMAX_DELAY)) {
    while (!SD.begin(SD_CS_PIN)) {
      xSemaphoreGive(sdMutex);
      display.clearDisplay();
      display.fillRect(0, 0, 128, 16, SSD1306_WHITE);
      display.setTextColor(SSD1306_BLACK, SSD1306_WHITE);
      display.setCursor(4, 4);
      display.print("!! SD CARD MISSING !!");
      display.setTextColor(SSD1306_WHITE, SSD1306_BLACK);
      display.setCursor(0, 24);
      display.print("Insert SD Card Now");
      display.setCursor(0, 36);
      display.print("System is HALTED.");
      display.setCursor(0, 50);
      display.print("Retrying...");
      display.display();
      tone(BUZZER, 200, 500);
      delay(600);
      tone(BUZZER, 200, 500);
      delay(600);
      xSemaphoreTake(sdMutex, portMAX_DELAY);
    }
    sdReady = true;
    sdTotalMB = SD.cardSize() / (1024 * 1024);
    if (!SD.exists("/master_log.txt")) {
      File f = SD.open("/master_log.txt", FILE_WRITE);
      if (f)
        f.close();
    }
    if (!SD.exists("/unsent.txt")) {
      File f = SD.open("/unsent.txt", FILE_WRITE);
      if (f)
        f.close();
    }
    countUnsentRecords();
    xSemaphoreGive(sdMutex);
  }

  // Preferences & Offline Cache
  preferences.begin("smssys", false);
  cachedMasterRFID = preferences.getString("master", "");
  cachedPortalOpen = preferences.getInt("portal", 1);
  masterFingerSlot = preferences.getInt("mfSlot", 200);
  loadOfflineCache();

  // Fingerprint Sensor
  mySerial.begin(57600, SERIAL_8N1, RX2_PIN, TX2_PIN);
  if (!finger.verifyPassword()) {
    Serial.println("[ERROR] Fingerprint sensor not found! Check TX/RX wiring!");
    display.clearDisplay();
    display.setTextColor(SSD1306_WHITE, SSD1306_BLACK);
    display.setTextSize(1);
    display.setCursor(0, 20);
    display.print("FINGERPRINT ERROR!");
    display.setCursor(0, 35);
    display.print("Check AS608 Wiring");
    display.setCursor(0, 48);
    display.print("(TX/RX may be swapped)");
    display.display();
    while (1) { delay(100); }
  }
  finger.getParameters(); // Required by some AS608 modules to sync packet sizes

  // NFC
  nfc.begin();
  if (nfc.getFirmwareVersion())
    nfc.SAMConfig();

  // I2S Microphone
  setupI2S();
  Serial.println(
      "[I2S] INMP441 microphone initialized on SD=14, WS=27, SCK=25");

  // Pre-load whitelist from SD into RAM — fingerprint attendance works
  // instantly from power-on, even before WiFi/server connects.
  loadWhitelistFromSD();

  showBootSequence();

  // =================== FreeRTOS Task Creation ====================
  // Core 0: Network + Voice Recording (both run here cooperatively)
  xTaskCreatePinnedToCore(networkTask, "NetworkTask", 12000, NULL, 1, NULL, 0);
  xTaskCreatePinnedToCore(voiceTask, "VoiceTask", 8000, NULL, 2, NULL, 0);
  // Core 1: Sensor reading + OLED UI
  xTaskCreatePinnedToCore(sensorUITask, "SensorUITask", 10000, NULL, 1, NULL,
                          1);
  // Core 0: SD Card watchdog (high priority)
  xTaskCreatePinnedToCore(sdGuardTask, "SDGuardTask", 4000, NULL, 3, NULL, 0);
}

// ========================= CORE 0: Voice Recording Task ============
// Records 9 seconds of I2S audio directly to SD card in 1KB chunks.
// No large RAM buffer required — works on any ESP32.
void voiceTask(void *parameter) {
  for (;;) {
    // Block until Core 1 signals a voice capture request
    EventBits_t bits = xEventGroupWaitBits(voiceEventGroup,
                                           VOICE_REQUEST_BIT | VOICE_ABORT_BIT,
                                           pdTRUE,  // clear bits on exit
                                           pdFALSE, // wait for ANY bit
                                           portMAX_DELAY);

    if (bits & VOICE_ABORT_BIT) {
      voiceFileSaved = false;
      xEventGroupSetBits(voiceEventGroup, VOICE_DONE_BIT);
      continue;
    }
    if (!(bits & VOICE_REQUEST_BIT))
      continue;

    voiceFileSaved = false;

    // Grab SD mutex for the full recording window (9 s)
    if (!xSemaphoreTake(sdMutex, pdMS_TO_TICKS(2000))) {
      Serial.println("[Voice] Could not take sdMutex — skipping recording.");
      xEventGroupSetBits(voiceEventGroup, VOICE_DONE_BIT);
      continue;
    }

    // Remove previous temp file and open new one
    if (SD.exists(VOICE_TEMP_PATH))
      SD.remove(VOICE_TEMP_PATH);
    File wavFile = SD.open(VOICE_TEMP_PATH, FILE_WRITE);
    if (!wavFile) {
      xSemaphoreGive(sdMutex);
      Serial.println("[Voice] Failed to open voice_temp.wav on SD!");
      xEventGroupSetBits(voiceEventGroup, VOICE_DONE_BIT);
      continue;
    }

    // Write a placeholder 44-byte WAV header (will be fixed after recording)
    uint8_t header[44] = {0};
    wavFile.write(header, 44);

    // ── Record audio in 1KB chunks ──
    static uint8_t chunk[1024]; // static: lives in BSS, not stack
    int remaining = VOICE_DATA_BYTES;
    int bytesWritten = 0;
    bool aborted = false;
    size_t bytesRead = 0;

    Serial.println("[Voice] Recording to SD card (9 s)...");

    while (remaining > 0) {
      // Check for abort signal mid-record
      if (xEventGroupGetBits(voiceEventGroup) & VOICE_ABORT_BIT) {
        xEventGroupClearBits(voiceEventGroup, VOICE_ABORT_BIT);
        aborted = true;
        break;
      }

      int chunkSize = (remaining > 1024) ? 1024 : remaining;
      esp_err_t err =
          i2s_read(I2S_PORT, chunk, chunkSize, &bytesRead, pdMS_TO_TICKS(200));
      if (err != ESP_OK || bytesRead == 0)
        continue;

      wavFile.write(chunk, bytesRead);
      bytesWritten += bytesRead;
      remaining -= bytesRead;
    }

    // Fix the WAV header with actual data size
    if (!aborted && bytesWritten > 0) {
      wavFile.seek(0);
      buildWavHeader(header, bytesWritten);
      wavFile.write(header, 44);
    }
    wavFile.close();
    xSemaphoreGive(sdMutex);

    voiceFileSaved = !aborted && (bytesWritten > 0);
    Serial.printf("[Voice] Recording %s. Bytes to SD: %d\n",
                  voiceFileSaved ? "OK" : "ABORTED", bytesWritten);

    xEventGroupSetBits(voiceEventGroup, VOICE_DONE_BIT);
  }
}

// ========================= CORE 0: Network & Sync Task =============
void networkTask(void *parameter) {
  bool forceWifi = preferences.getBool("force_wifi", false);

  if (forceWifi) {
    preferences.putBool("force_wifi", false);
    triggerIndicatorSafe(false, "Wi-Fi Config\nPortal Open");
    currentMode = "wifi_setup";
    WiFiManager wm_setup;
    wm_setup.setConfigPortalTimeout(180);
    wm_setup.startConfigPortal("TEAM_X_SETUP", "admin123");
    triggerIndicatorSafe(true, "Wi-Fi Saved!\nRebooting...");
    vTaskDelay(1000 / portTICK_PERIOD_MS);
    ESP.restart();
  }

  WiFiManager wm;
  wm.setConnectTimeout(15);
  wm.setConfigPortalBlocking(false);
  if (!wm.autoConnect("TEAM_X_SETUP", "admin123")) {
  }

  unsigned long lastPollTime = 0;
  unsigned long lastSyncTime = 0;
  unsigned long lastCacheSync = 0;
  unsigned long lastSuccessfulPing = 0;
  unsigned long urlDiscoveryTime = 0;

  for (;;) {
    wm.process();
    bool wifiConnected = (WiFi.status() == WL_CONNECTED);

    if (wifiConnected) {
      static bool ntpConfigured = false;
      if (!ntpConfigured) {
        configTime(0, 0, "pool.ntp.org", "time.nist.gov"); // Fetch UTC time
        ntpConfigured = true;
      }
      
      if (ntpConfigured && rtcReady) {
        static bool rtcUpdatedFromNTP = false;
        if (!rtcUpdatedFromNTP) {
          struct tm timeinfo;
          // non-blocking check
          if (getLocalTime(&timeinfo, 10)) {
            if (timeinfo.tm_year > (2024 - 1900)) {
              rtc.adjust(DateTime(timeinfo.tm_year + 1900, timeinfo.tm_mon + 1, timeinfo.tm_mday, timeinfo.tm_hour, timeinfo.tm_min, timeinfo.tm_sec));
              rtcUpdatedFromNTP = true;
              Serial.println("[NTP] RTC successfully synced with internet time (UTC).");
            }
          }
        }
      }

      // mDNS auto-discovery
      if (serverBaseURL == "") {
        if (!MDNS.begin("esp32-node")) {
        }
        int n = MDNS.queryService("http", "tcp");
        bool found = false;
        for (int i = 0; i < n; i++) {
          if (MDNS.hostname(i).indexOf("esp32-node") >= 0 ||
              MDNS.port(i) == 5005) {
            serverBaseURL = "http://" + MDNS.address(i).toString() + ":" +
                            String(MDNS.port(i));
            found = true;
            break;
          }
        }
        if (!found)
          serverBaseURL = "http://esp32-node.local:5005";
        lastSuccessfulPing = 0;
        urlDiscoveryTime = millis();
      }

      isOnline =
          (serverBaseURL != "" && lastSuccessfulPing > urlDiscoveryTime &&
           (millis() - lastSuccessfulPing <= 10000));

      unsigned long timeSinceActivity =
          millis() - (lastSuccessfulPing > urlDiscoveryTime ? lastSuccessfulPing
                                                            : urlDiscoveryTime);
      if (timeSinceActivity > 30000 && serverBaseURL != "") {
        serverBaseURL = "";
        isOnline = false;
      }

      // Poll server status every 2 s
      if (millis() - lastPollTime > 2000 && serverBaseURL != "") {
        int httpCode = 0;
        String responsePayload =
            doHttpGet("/api/esp32/status", 2500, &httpCode);
        if (httpCode == 200) {
          isOnline = true;
          lastSuccessfulPing = millis();
          JsonDocument doc;
          deserializeJson(doc, responsePayload);
          String newMode = doc["mode"].as<String>();
          if (!doc["assign_id"].isNull())
            targetID = doc["assign_id"].as<int>();

          if (newMode == "force_audit") {
            pendingAudit = true;
            auditExpectedRFID = "SIMULATED";
            auditStartTime = millis();
            playAudio("error");
            if (xSemaphoreTake(displayMutex, portMAX_DELAY)) {
              display.clearDisplay();
              display.fillRect(0, 0, 128, 16, SSD1306_WHITE);
              display.setTextColor(SSD1306_BLACK, SSD1306_WHITE);
              display.setCursor(4, 4);
              display.print("SECURITY AUDIT");
              display.setTextColor(SSD1306_WHITE, SSD1306_BLACK);
              display.setCursor(0, 24);
              display.print("Proxy Suspected!");
              display.setCursor(0, 36);
              display.print("Verify Identity:");
              display.setCursor(0, 50);
              display.print("Place Fingerprint!");
              display.display();
              xSemaphoreGive(displayMutex);
            }
            currentMode = "attendance";
          } else if (newMode == "enroll_master_finger") {
            if (currentMode == "attendance")
              currentMode = "enroll_master_finger";
          } else if (newMode == "attendance") {
            bool activeOp =
                (currentMode == "enroll_fingerprint" ||
                 currentMode == "enroll_rfid" ||
                 currentMode == "enroll_master_finger" ||
                 currentMode == "capture_rfid" || currentMode == "delete_all" ||
                 currentMode == "wifi_setup");
            if (!activeOp)
              currentMode = "attendance";
          } else if (newMode != currentMode) {
            currentMode = newMode;
          }

          if (!doc["master_rfid"].isNull()) {
            String mRFID = doc["master_rfid"].as<String>();
            if (mRFID != cachedMasterRFID) {
              cachedMasterRFID = mRFID;
              preferences.putString("master", cachedMasterRFID);
            }
          }
          if (!doc["portal_open"].isNull()) {
            int pState = doc["portal_open"].as<int>();
            if (pState != cachedPortalOpen) {
              cachedPortalOpen = pState;
              preferences.putInt("portal", cachedPortalOpen);
            }
          }
          if (!doc["timestamp"].isNull() && rtcReady) {
            long serverEpoch = doc["timestamp"].as<long>();
            DateTime now = rtc.now();
            if (abs((long)now.unixtime() - serverEpoch) > 60) {
              rtc.adjust(DateTime(serverEpoch));
            }
          }
        }
        lastPollTime = millis();
      }

      // Whitelist refresh (24 h or missing)
      if (serverBaseURL != "") {
        bool needsUpdate = false;
        if (whitelistCount == 0 && !sdReady)
          needsUpdate = true;
        else if (sdReady && millis() - lastWhitelistFetch > 86400000UL)
          needsUpdate = true;
        else if (sdReady && !SD.exists("/whitelist.txt"))
          needsUpdate = true;

        if (needsUpdate || lastWhitelistFetch == 0) {
          fetchWhitelist();
          fetchTodayAttendance();
          lastWhitelistFetch = millis();
          lastCacheSync = millis();
        }
      }

      // Attendance cache re-sync every 5 minutes
      if (serverBaseURL != "" && millis() - lastCacheSync > 300000UL) {
        fetchTodayAttendance();
        lastCacheSync = millis();
      }

      // Batch sync offline records — only when online AND there is actually unsent data
      if (isOnline && sdReady && unsentCount > 0 &&
          (millis() - lastSyncTime > 10000)) {
        isSyncing = true;
        updateDisplaySafe();

        const int BATCH_SIZE = 20;
        String batchData[BATCH_SIZE];
        int batchCount = 0;
        bool readSuccess = false;

        if (xSemaphoreTake(sdMutex, portMAX_DELAY)) {
          File file = SD.open("/unsent.txt", FILE_READ);
          if (file) {
            while (file.available() && batchCount < BATCH_SIZE) {
              String payload = file.readStringUntil('\n');
              payload.trim();
              if (payload.length() > 0)
                batchData[batchCount++] = payload;
            }
            file.close();
            readSuccess = true;
          }
          xSemaphoreGive(sdMutex);
        }

        bool batchSuccess = false;
        if (readSuccess && batchCount > 0) {
          String jsonArray = "[";
          for (int i = 0; i < batchCount; i++) {
            jsonArray += batchData[i];
            if (i < batchCount - 1)
              jsonArray += ",";
          }
          jsonArray += "]";
          int httpCode = 0;
          doHttpPost("/api/esp32/attendance/batch", jsonArray, 4000, &httpCode);
          if (httpCode == 200)
            batchSuccess = true;
        }

        if (batchSuccess) {
          if (xSemaphoreTake(sdMutex, portMAX_DELAY)) {
            File file = SD.open("/unsent.txt", FILE_READ);
            File tmpFile = SD.open("/temp.txt", FILE_WRITE);
            if (file && tmpFile) {
              int skipped = 0;
              while (file.available() && skipped < batchCount) {
                String payload = file.readStringUntil('\n');
                payload.trim();
                if (payload.length() > 0)
                  skipped++;
              }
              while (file.available()) {
                String payload = file.readStringUntil('\n');
                payload.trim();
                if (payload.length() > 0)
                  tmpFile.println(payload);
              }
              file.close();
              tmpFile.close();
              SD.remove("/unsent.txt");
              SD.rename("/temp.txt", "/unsent.txt");
              countUnsentRecords();
            } else {
              if (file)
                file.close();
              if (tmpFile)
                tmpFile.close();
            }
            xSemaphoreGive(sdMutex);
          }
        }

        isSyncing = false;
        lastSyncTime = millis();
        if (unsentCount == 0 && offlineCacheCount > 0) {
          playAudio("success");
          offlineCacheCount = 0;
        }
      } else if (millis() - lastSyncTime > 10000) {
        // Nothing to sync (offline or no unsent records) — reset timer quietly
        lastSyncTime = millis();
      }

    }

    // SD remount retry
    if (!sdReady && (millis() - lastSyncTime > 15000)) {
      if (xSemaphoreTake(sdMutex, portMAX_DELAY)) {
        if (SD.begin(SD_CS_PIN)) {
          sdReady = true;
          sdTotalMB = SD.cardSize() / (1024 * 1024);
          countUnsentRecords();
          playAudio("success");
        }
        xSemaphoreGive(sdMutex);
      }
      lastSyncTime = millis();
    }

    // ── Reconnect Detection ─────────────────────────────────────────
    // When the server comes back online after being offline, immediately
    // refresh whitelist and today's attendance cache so the device has
    // the latest student data and duplication prevention is accurate.
    if (isOnline && !wasOnline) {
      Serial.println("[Network] Server reconnected — refreshing whitelist & "
                     "attendance cache.");
      fetchWhitelist();
      fetchTodayAttendance();
      lastWhitelistFetch = millis();
      lastCacheSync = millis();
    }
    wasOnline = isOnline;
    // ────────────────────────────────────────────────────────────────

    vTaskDelay(100 / portTICK_PERIOD_MS);
  }
}

// ========================= CORE 0: SD Guard Task ===================
void sdGuardTask(void *parameter) {
  static int consecutiveFailCycles = 0;   // must fail N cycles in a row
  const  int FAIL_THRESHOLD = 3;          // 3 × 15 s = 45 s of failures needed
  const  int CHECK_INTERVAL = 15000;      // ms between checks

  for (;;) {
    vTaskDelay(CHECK_INTERVAL / portTICK_PERIOD_MS);

    // Try to acquire the mutex with a short timeout.
    // If it's busy (voice/sync task using SD), skip — card is clearly present.
    if (!xSemaphoreTake(sdMutex, pdMS_TO_TICKS(200))) {
      // mutex busy = someone else is using SD = card must be OK
      consecutiveFailCycles = 0;
      continue;
    }

    // Small settle delay in case we grabbed the mutex right after a big write
    delay(20);

    // --- Primary check: card type (very fast, non-destructive) ---
    bool cardPresent = (SD.cardType() != CARD_NONE);

    // --- Secondary check: open root dir (only if cardType looks OK) ---
    if (cardPresent) {
      File root = SD.open("/");
      if (root) {
        root.close();
        consecutiveFailCycles = 0; // card is healthy
      } else {
        // SD.open failed but cardType said card is there — could be transient
        // Retry once after a brief settle
        delay(100);
        File root2 = SD.open("/");
        if (root2) {
          root2.close();
          consecutiveFailCycles = 0;
        } else {
          consecutiveFailCycles++;
          Serial.printf("[SDGuard] Open / failed (cycle %d/%d)\n",
                        consecutiveFailCycles, FAIL_THRESHOLD);
        }
      }
    } else {
      consecutiveFailCycles++;
      Serial.printf("[SDGuard] cardType = NONE (cycle %d/%d)\n",
                    consecutiveFailCycles, FAIL_THRESHOLD);
    }

    xSemaphoreGive(sdMutex);

    // Only declare error after N consecutive bad cycles (not a one-off glitch)
    if (consecutiveFailCycles < FAIL_THRESHOLD) {
      continue;
    }

    // ── Card genuinely missing or failed ──────────────────────────
    consecutiveFailCycles = 0; // reset so recovery loop can re-detect
    sdReady = false;
    Serial.println("[SDGuard] SD card confirmed FAILED — showing error.");

    while (true) {
      if (xSemaphoreTake(displayMutex, pdMS_TO_TICKS(100))) {
        display.clearDisplay();
        display.fillRect(0, 0, 128, 16, SSD1306_WHITE);
        display.setTextColor(SSD1306_BLACK, SSD1306_WHITE);
        display.setCursor(4, 4);
        display.print("!! SD CARD ERROR !!");
        display.setTextColor(SSD1306_WHITE, SSD1306_BLACK);
        display.setCursor(0, 20);
        display.print("INSERT SD CARD NOW");
        display.setCursor(0, 32);
        display.print("Logs cannot be saved");
        display.setCursor(0, 44);
        display.print("Scanning still works");
        display.display();
        xSemaphoreGive(displayMutex);
      }
      tone(BUZZER, 300, 400);
      vTaskDelay(1000 / portTICK_PERIOD_MS);

      // Try recovery
      if (xSemaphoreTake(sdMutex, pdMS_TO_TICKS(500))) {
        if (SD.begin(SD_CS_PIN)) {
          sdReady = true;
          if (!SD.exists("/master_log.txt")) {
            File f = SD.open("/master_log.txt", FILE_WRITE);
            if (f) f.close();
          }
          if (!SD.exists("/unsent.txt")) {
            File f = SD.open("/unsent.txt", FILE_WRITE);
            if (f) f.close();
          }
          countUnsentRecords();
          xSemaphoreGive(sdMutex);
          playAudio("success");
          Serial.println("[SDGuard] SD card recovered!");
          break;
        }
        xSemaphoreGive(sdMutex);
      }
    }
  }
}


// ========================= CORE 1: Sensor & UI Task ================
void sensorUITask(void *parameter) {
  for (;;) {
    // Manual serial commands
    if (Serial.available() > 0) {
      String cmd = Serial.readStringUntil('\n');
      cmd.trim();
      if (cmd == "SETUP_WIFI" || cmd == "setup_wifi") {
        Serial.println("Manual override: restarting into Wi-Fi setup mode.");
        preferences.putBool("force_wifi", true);
        triggerIndicatorSafe(false, "Rebooting to\nWi-Fi Setup");
        ESP.restart();
      }
    }

    // Animate display every 800 ms
    if (millis() - lastAnimTime > 800) {
      animFrame = !animFrame;
      lastAnimTime = millis();
      updateDisplaySafe();

      // Audit alarm beep
      if (pendingAudit) {
        tone(BUZZER, animFrame ? 800 : 400, 400);
      }
    }

    if (!isSyncing) {
      if (currentMode == "attendance")
        handleAttendance();
      else if (currentMode == "enroll_fingerprint")
        handleFingerprintEnrollment();
      else if (currentMode == "enroll_rfid")
        handleRFIDEnrollment();
      else if (currentMode == "capture_rfid")
        handleCaptureRfid();
      else if (currentMode == "enroll_master_finger")
        handleMasterFingerEnrollment();
      else if (currentMode == "delete_all")
        handleDeleteAll();
      else if (currentMode == "test_record")
        handleTestRecord();
    }

    vTaskDelay(50 / portTICK_PERIOD_MS);
  }
}

// ========================= Voice Upload Helper =====================
// Called from Core 1 after voice recording completes.
// Streams the WAV file from SD card to /api/esp32/voice_submit.
bool uploadVoiceAndGetDecision(const char *rfidUid, bool &outProxy,
                               bool &outWrongPhrase, String &outMatchedName,
                               float &outScore) {
  outProxy = false;
  outMatchedName = "";
  outScore = 0.0f;

  if (!isOnline || serverBaseURL == "") {
    Serial.println("[Voice] Skipping upload — server offline.");
    return false;
  }
  if (!voiceFileSaved) {
    Serial.println("[Voice] Skipping upload — no file saved.");
    return false;
  }

  String responseOut;
  bool ok = doHttpPostMultipart(
      "/api/esp32/voice_submit", rfidUid,
      30000, // 30 s timeout for upload + SpeechBrain inference
      responseOut);

  if (!ok || responseOut.length() == 0) {
    Serial.println("[Voice] Upload failed or empty response.");
    return false;
  }

  JsonDocument res;
  DeserializationError err = deserializeJson(res, responseOut);
  if (err) {
    Serial.printf("[Voice] JSON parse error: %s\n", err.c_str());
    return false;
  }

  outProxy = res["proxy"].as<bool>();
  outWrongPhrase = res["wrong_phrase"].as<bool>();

  if (outWrongPhrase) {
    Serial.println("[Voice] Wrong phrase detected by ASR. Retrying...");
  } else if (outProxy) {
    outMatchedName = res["matched_student_name"].as<String>();
    outScore = res["score"].as<float>();
    Serial.printf("[Voice] PROXY! Matched: %s at %.1f%%\n",
                  outMatchedName.c_str(), outScore);
  } else {
    outScore = res["best_score"].as<float>();
    Serial.printf("[Voice] OK. Best score: %.1f%%\n", outScore);
  }
  return true;
}

// ========================= Multipart HTTP POST (SD streaming) ======
// Streams WAV from SD card via raw WiFiClient — zero large malloc needed.
bool doHttpPostMultipart(String endpoint, const char *rfidUid, int timeoutMs,
                         String &responseOut) {
  if (serverBaseURL == "" || WiFi.status() != WL_CONNECTED)
    return false;

  // --- Get WAV file size from SD ---
  int wavSize = 0;
  if (!xSemaphoreTake(sdMutex, pdMS_TO_TICKS(2000)))
    return false;
  if (!SD.exists(VOICE_TEMP_PATH)) {
    xSemaphoreGive(sdMutex);
    Serial.println("[Voice] voice_temp.wav not found on SD!");
    return false;
  }
  File wavFile = SD.open(VOICE_TEMP_PATH, FILE_READ);
  if (!wavFile) {
    xSemaphoreGive(sdMutex);
    return false;
  }
  wavSize = wavFile.size();
  // Keep file open — we'll stream from it below

  // --- Build multipart parts ---
  String boundary = "ESP32VoiceBoundary";
  String part1 = "--" + boundary + "\r\n";
  part1 += "Content-Disposition: form-data; name=\"rfid_uid\"\r\n\r\n";
  part1 += String(rfidUid) + "\r\n";
  part1 += "--" + boundary + "\r\n";
  part1 += "Content-Disposition: form-data; name=\"audio\"; "
           "filename=\"voice.wav\"\r\n";
  part1 += "Content-Type: audio/wav\r\n\r\n";
  String part3 = "\r\n--" + boundary + "--\r\n";
  int totalLen = part1.length() + wavSize + part3.length();

  // --- Parse host & port from serverBaseURL ---
  String hostStr = serverBaseURL;
  hostStr.replace("http://", "");
  int colonPos = hostStr.indexOf(':');
  int port = 5005;
  if (colonPos > 0) {
    port = hostStr.substring(colonPos + 1).toInt();
    hostStr = hostStr.substring(0, colonPos);
  }

  // --- Open raw TCP connection ---
  WiFiClient client;
  client.setTimeout(timeoutMs / 1000 + 5);
  if (!client.connect(hostStr.c_str(), port)) {
    wavFile.close();
    xSemaphoreGive(sdMutex);
    Serial.println("[Voice] TCP connect failed!");
    return false;
  }

  // --- Send HTTP request line + headers ---
  client.print("POST " + endpoint + " HTTP/1.1\r\n");
  client.print("Host: " + hostStr + ":" + String(port) + "\r\n");
  client.print("Content-Type: multipart/form-data; boundary=" + boundary +
               "\r\n");
  client.print("Content-Length: " + String(totalLen) + "\r\n");
  client.print("Connection: close\r\n\r\n");

  // --- Send multipart preamble ---
  client.print(part1);

  // --- Stream WAV from SD in 512-byte chunks ---
  static uint8_t txBuf[512];
  while (wavFile.available()) {
    int n = wavFile.read(txBuf, sizeof(txBuf));
    if (n > 0)
      client.write(txBuf, n);
  }
  wavFile.close();
  xSemaphoreGive(sdMutex);

  // --- Send multipart epilogue ---
  client.print(part3);
  client.flush();

  // --- Read HTTP response ---
  unsigned long deadline = millis() + (unsigned long)timeoutMs;
  bool headersEnded = false;
  int statusCode = 0;
  responseOut = "";

  while (client.connected() && millis() < deadline) {
    while (client.available() && millis() < deadline) {
      String line = client.readStringUntil('\n');
      if (!headersEnded) {
        if (statusCode == 0 && line.startsWith("HTTP/")) {
          statusCode = line.substring(9, 12).toInt();
        }
        if (line == "\r" || line.length() <= 1) {
          headersEnded = true;
        }
      } else {
        responseOut += line;
      }
    }
    if (headersEnded && responseOut.length() > 0)
      break;
    delay(10);
  }
  client.stop();

  Serial.printf("[Voice] SD multipart upload → HTTP %d (%d bytes response)\n",
                statusCode, responseOut.length());
  return (statusCode == 200);
}

// ========================= fetchWhitelist ==========================
void fetchWhitelist() {
  if (serverBaseURL == "" || isSyncing)
    return;

  HTTPClient http;
  http.begin(serverBaseURL + "/api/esp32/whitelist");
  int httpCode = http.GET();
  if (httpCode == 200) {
    if (sdReady && xSemaphoreTake(sdMutex, portMAX_DELAY)) {
      File file = SD.open("/whitelist.txt", FILE_WRITE);
      if (file) {
        JsonDocument doc;
        if (!deserializeJson(doc, http.getStream())) {
          for (JsonObject elem : doc.as<JsonArray>()) {
            file.print(elem["s"].as<int>());
            file.print(",");
            file.print(elem["f"].as<int>());
            file.print(",");
            file.print(elem["r"].as<String>());
            file.print(",");
            file.print(elem["n"].as<String>());
            file.print(",");
            file.println(elem["b"].as<int>());
          }
        }
        file.close();
      }
      xSemaphoreGive(sdMutex);
    }
    // RAM fallback
    {
      HTTPClient httpFall;
      httpFall.begin(serverBaseURL + "/api/esp32/whitelist");
      if (httpFall.GET() == 200) {
        JsonDocument doc;
        if (!deserializeJson(doc, httpFall.getStream())) {
          whitelistCount = 0;
          for (JsonObject elem : doc.as<JsonArray>()) {
            if (whitelistCount >= MAX_STUDENTS)
              break;
            whitelist[whitelistCount].student_id = elem["s"];
            whitelist[whitelistCount].fingerprint_id = elem["f"];
            String r = elem["r"].as<String>();
            snprintf(whitelist[whitelistCount].rfid_uid, 15, "%s", r.c_str());
            String n = elem["n"].as<String>();
            snprintf(whitelist[whitelistCount].name, 16, "%s", n.c_str());
            whitelist[whitelistCount].is_blacklisted =
                (elem["b"].as<int>() == 1);
            whitelistCount++;
          }
        }
      }
      httpFall.end();
    }
  }
  http.end();
}

// ========================= fetchTodayAttendance ====================
void fetchTodayAttendance() {
  if (serverBaseURL == "" || isSyncing)
    return;
  HTTPClient http;
  http.begin(serverBaseURL + "/api/esp32/today_attendance");
  int httpCode = http.GET();
  if (httpCode == 200) {
    String payload = http.getString();
    JsonDocument doc;
    if (!deserializeJson(doc, payload)) {
      offlineCacheCount = 0;
      String todayDate = getLocalDateStr();
      offlineCacheDate = todayDate;
      if (sdReady && xSemaphoreTake(sdMutex, portMAX_DELAY)) {
        if (SD.exists("/today_cache.txt"))
          SD.remove("/today_cache.txt");
        xSemaphoreGive(sdMutex);
      }
      for (JsonVariant elem : doc.as<JsonArray>()) {
        addToOfflineCache(elem.as<int>());
      }
    }
  }
  http.end();
}

// ========================= loadWhitelistFromSD =====================
// Reads /whitelist.txt from SD into the RAM whitelist[] array.
// Called once at boot so fingerprint matching works instantly offline
// without waiting for a server connection.
void loadWhitelistFromSD() {
  if (!sdReady)
    return;
  if (!xSemaphoreTake(sdMutex, pdMS_TO_TICKS(2000)))
    return;
  if (!SD.exists("/whitelist.txt")) {
    xSemaphoreGive(sdMutex);
    return;
  }
  File file = SD.open("/whitelist.txt", FILE_READ);
  if (!file) {
    xSemaphoreGive(sdMutex);
    return;
  }
  whitelistCount = 0;
  while (file.available() && whitelistCount < MAX_STUDENTS) {
    String line = file.readStringUntil('\n');
    line.trim();
    if (line.length() == 0)
      continue;
    // Format: student_id,fingerprint_id,rfid_uid,name,blacklisted
    int c1 = line.indexOf(',');
    int c2 = line.indexOf(',', c1 + 1);
    int c3 = line.indexOf(',', c2 + 1);
    int c4 = line.indexOf(',', c3 + 1);
    if (c1 < 0 || c2 < 0 || c3 < 0)
      continue; // malformed line
    whitelist[whitelistCount].student_id = line.substring(0, c1).toInt();
    whitelist[whitelistCount].fingerprint_id =
        line.substring(c1 + 1, c2).toInt();
    String ruid = line.substring(c2 + 1, c3);
    ruid.trim();
    snprintf(whitelist[whitelistCount].rfid_uid, sizeof(whitelist[0].rfid_uid),
             "%s", ruid.c_str());
    String name;
    if (c4 > c3) {
      name = line.substring(c3 + 1, c4);
      name.trim();
      whitelist[whitelistCount].is_blacklisted =
          (line.substring(c4 + 1).toInt() == 1);
    } else {
      name = line.substring(c3 + 1);
      name.trim();
      whitelist[whitelistCount].is_blacklisted = false;
    }
    snprintf(whitelist[whitelistCount].name, sizeof(whitelist[0].name), "%s",
             name.c_str());
    whitelistCount++;
  }
  file.close();
  xSemaphoreGive(sdMutex);
  Serial.printf("[Boot] Loaded %d students from SD whitelist into RAM.\n",
                whitelistCount);
}

          display.drawCircle(110, 40, 8, SSD1306_WHITE);
      } else {
        DateTime utcD = rtcReady ? rtc.now() : DateTime(2000, 1, 1);
        DateTime localD = DateTime(utcD.unixtime() + TIMEZONE_OFFSET_SEC);
        char dateBuf[12];
        snprintf(dateBuf, sizeof(dateBuf), "%04d-%02d-%02d", localD.year(),
                 localD.month(), localD.day());
        display.print("Date: ");
        display.print(dateBuf);
        if (!isOnline) {
          // ── Offline mode: show fingerprint-only notice ──
          display.setCursor(0, 30);
          display.println("** OFFLINE MODE **");
          display.setCursor(0, 42);
          display.print("Fingerprint Only");
          if (unsentCount > 0) {
            display.setCursor(0, 54);
            display.print("Unsent: ");
            display.print(unsentCount);
          }
        } else {
          // ── Online mode: normal status ──
          display.setCursor(0, 32);
          if (unsentCount == 0)
            display.print("All Data Synced");
          else {
            display.print("Unsent Logs: ");
            display.print(unsentCount);
          }
          display.setCursor(0, 48);
          if (animFrame) {
            display.print(">> SCAN NOW >>");
            display.fillCircle(115, 50, 4, SSD1306_WHITE);
          } else {
            display.print("   SCAN NOW   ");
            display.drawCircle(115, 50, 5, SSD1306_WHITE);
          }
        }
      }
    } else {
      display.setCursor(0, 20);
      display.println("MODE:");
      display.println(currentMode);
    }
    display.display();
    xSemaphoreGive(displayMutex);
  }
}

}

// ========================= Time Helpers ============================
String getISOTime() {
  if (!rtcReady)
    return "1970-01-01T00:00:00Z";
  DateTime now = rtc.now();
  char buf[25];
  snprintf(buf, sizeof(buf), "%04d-%02d-%02dT%02d:%02d:%02dZ", now.year(),
           now.month(), now.day(), now.hour(), now.minute(), now.second());
  return String(buf);
}

String getLocalDateStr() {
  if (!rtcReady)
    return "1970-01-01";
  DateTime localNow = DateTime(rtc.now().unixtime() + TIMEZONE_OFFSET_SEC);
  char buf[11];
  snprintf(buf, sizeof(buf), "%04d-%02d-%02d", localNow.year(),
           localNow.month(), localNow.day());
  return String(buf);
}

// ========================= Offline Cache ===========================
void loadOfflineCache() {
  if (!sdReady)
    return;
  String todayDate = getLocalDateStr();
  if (todayDate == "1970-01-01")
    return;
  offlineCacheCount = 0;
  offlineCacheDate = todayDate;
  if (xSemaphoreTake(sdMutex, portMAX_DELAY)) {
    if (SD.exists("/today_cache.txt")) {
      File f = SD.open("/today_cache.txt", FILE_READ);
      if (f) {
        String fileDateLine = f.readStringUntil('\n');
        fileDateLine.trim();
        if (fileDateLine == todayDate) {
          while (f.available() && offlineCacheCount < MAX_OFFLINE_CACHE) {
            String line = f.readStringUntil('\n');
            line.trim();
            if (line.length() > 0)
              offlineCache[offlineCacheCount++] = line;
          }
        } else {
          f.close();
          SD.remove("/today_cache.txt");
          xSemaphoreGive(sdMutex);
          return;
        }
        f.close();
      }
    }
    xSemaphoreGive(sdMutex);
  }
}

bool isDuplicateOfflineScan(int studentId) {
  String todayDate = getLocalDateStr();
  if (todayDate == "1970-01-01")
    return false;
  if (todayDate != offlineCacheDate) {
    offlineCacheCount = 0;
    offlineCacheDate = todayDate;
    if (sdReady && xSemaphoreTake(sdMutex, portMAX_DELAY)) {
      if (SD.exists("/today_cache.txt"))
        SD.remove("/today_cache.txt");
      xSemaphoreGive(sdMutex);
    }
  }
  String searchKey = String(studentId);
  for (int i = 0; i < offlineCacheCount; i++) {
    if (offlineCache[i] == searchKey)
      return true;
  }
  return false;
}

void addToOfflineCache(int studentId) {
  String todayDate = getLocalDateStr();
  if (todayDate == "1970-01-01")
    return;
  if (todayDate != offlineCacheDate) {
    offlineCacheCount = 0;
    offlineCacheDate = todayDate;
    if (sdReady && xSemaphoreTake(sdMutex, portMAX_DELAY)) {
      if (SD.exists("/today_cache.txt"))
        SD.remove("/today_cache.txt");
      xSemaphoreGive(sdMutex);
    }
  }
  if (offlineCacheCount < MAX_OFFLINE_CACHE) {
    String idStr = String(studentId);
    bool exists = false;
    for (int i = 0; i < offlineCacheCount; i++) {
      if (offlineCache[i] == idStr) {
        exists = true;
        break;
      }
    }
    if (!exists) {
      offlineCache[offlineCacheCount++] = idStr;
      if (sdReady && xSemaphoreTake(sdMutex, portMAX_DELAY)) {
        bool fileExists = SD.exists("/today_cache.txt");
        File f = SD.open("/today_cache.txt", FILE_APPEND);
        if (f) {
          if (!fileExists)
            f.println(todayDate);
          f.println(idStr);
          f.close();
        }
        xSemaphoreGive(sdMutex);
      }
    }
  }
}

// ========================= HTTP Helpers ============================
String doHttpGet(String endpoint, int timeoutMs, int *outHttpCode) {
  String payload = "";
  if (serverBaseURL == "" || WiFi.status() != WL_CONNECTED)
    return payload;
  HTTPClient http;
  http.begin(serverBaseURL + endpoint);
  http.setTimeout(timeoutMs);
  int httpCode = http.GET();
  if (outHttpCode)
    *outHttpCode = httpCode;
  if (httpCode == 200)
    payload = http.getString();
  http.end();
  return payload;
}

String doHttpPost(String endpoint, String jsonPayload, int timeoutMs,
                  int *outHttpCode) {
  String payload = "";
  if (serverBaseURL == "" || WiFi.status() != WL_CONNECTED)
    return payload;
  HTTPClient http;
  http.begin(serverBaseURL + endpoint);
  http.addHeader("Content-Type", "application/json");
  http.setTimeout(timeoutMs);
  int httpCode = http.POST(jsonPayload);
  if (outHttpCode)
    *outHttpCode = httpCode;
  if (httpCode == 200)
    payload = http.getString();
  http.end();
  return payload;
}

int getWhitelistIndexByRFID(String rfid) {
  for (int i = 0; i < whitelistCount; i++) {
    if (String(whitelist[i].rfid_uid) == rfid)
      return i;
  }
  return -1;
}

// ========================= processAttendanceData ===================
void processAttendanceData(String sensorType, String identifier) {
  String timestamp = getISOTime();

  // Master Card check
  if (sensorType == "rfid" && cachedMasterRFID != "" &&
      identifier == cachedMasterRFID) {
    cachedPortalOpen = (cachedPortalOpen == 1) ? 0 : 1;
    preferences.putInt("portal", cachedPortalOpen);
    String actionMsg =
        (cachedPortalOpen == 1) ? "PORTAL OPENED" : "PORTAL CLOSED";
    triggerIndicatorSafe(true, actionMsg);
    playAudio("startup");
    JsonDocument doc;
    doc["sensor"] = "rfid";
    doc["uid"] = identifier;
    doc["timestamp"] = timestamp;
    String jsonString;
    serializeJson(doc, jsonString);
    if (isOnline && serverBaseURL != "") {
      HTTPClient http;
      http.begin(serverBaseURL + "/api/esp32/attendance");
      http.addHeader("Content-Type", "application/json");
      http.setTimeout(4000);
      http.POST(jsonString);
    } else {
      logDataToSD(jsonString);
      if (xSemaphoreTake(sdMutex, portMAX_DELAY)) {
        File file = SD.open("/unsent.txt", FILE_APPEND);
        if (file) {
          file.println(jsonString);
          file.close();
        }
        xSemaphoreGive(sdMutex);
      }
      countUnsentRecords();
    }
    return;
  }

  // Portal check
  if (cachedPortalOpen == 0) {
    playAudio("error");
    triggerIndicatorSafe(false, "Portal Closed");
    return;
  }

  // Resolve student ID from whitelist (RAM first, then SD)
  int resolvedStudentId = -1;
  for (int i = 0; i < whitelistCount; i++) {
    if (sensorType == "fingerprint" &&
        whitelist[i].fingerprint_id == identifier.toInt()) {
      resolvedStudentId = whitelist[i].student_id;
      break;
    }
    if (sensorType == "rfid" && String(whitelist[i].rfid_uid) == identifier) {
      resolvedStudentId = whitelist[i].student_id;
      break;
    }
  }
  if (resolvedStudentId == -1 && sdReady) {
    if (xSemaphoreTake(sdMutex, portMAX_DELAY)) {
      File file = SD.open("/whitelist.txt", FILE_READ);
      if (file) {
        while (file.available()) {
          String line = file.readStringUntil('\n');
          line.trim();
          if (line.length() == 0)
            continue;
          int c1 = line.indexOf(',');
          int c2 = line.indexOf(',', c1 + 1);
          if (c1 > 0 && c2 > c1) {
            int s_id = line.substring(0, c1).toInt();
            int f_id = line.substring(c1 + 1, c2).toInt();
            String r_uid = line.substring(c2 + 1);
            r_uid.trim();
            if (sensorType == "fingerprint" && f_id == identifier.toInt()) {
              resolvedStudentId = s_id;
              break;
            }
            if (sensorType == "rfid" && r_uid == identifier) {
              resolvedStudentId = s_id;
              break;
            }
          }
        }
        file.close();
      }
      xSemaphoreGive(sdMutex);
    }
  }

  // Offline blacklist check
  if (resolvedStudentId != -1) {
    bool is_blocked = false;
    int idx = getWhitelistIndexByRFID(identifier);
    if (idx >= 0 && whitelist[idx].is_blacklisted)
      is_blocked = true;
    if (!is_blocked && sensorType == "fingerprint") {
      for (int i = 0; i < whitelistCount; i++) {
        if (whitelist[i].student_id == resolvedStudentId &&
            whitelist[i].is_blacklisted) {
          is_blocked = true;
          break;
        }
      }
    }
    if (is_blocked) {
      playAudio("error");
      triggerIndicatorSafe(false, "CARD BLOCKED\nProxy Flagged!");
      return;
    }
  }

  // Offline duplicate check
  if (resolvedStudentId != -1 && isDuplicateOfflineScan(resolvedStudentId)) {
    playAudio("duplicate");
    triggerIndicatorSafe(false, "Already Present\nToday!");
    return;
  }

  // Online mode
  if (isOnline && serverBaseURL != "") {
    JsonDocument doc;
    doc["sensor"] = sensorType;
    if (sensorType == "fingerprint")
      doc["id"] = identifier.toInt();
    else
      doc["uid"] = identifier;
    doc["timestamp"] = timestamp;
    if (sensorType == "rfid") {
      doc["ai_dwell"] = lastDwellTime;
      doc["ai_inter"] = lastInterScanTime;
      doc["ai_prediction"] = "NORMAL";
    }
    String jsonString;
    serializeJson(doc, jsonString);

    int httpCode = 0;
    String responsePayload =
        doHttpPost("/api/esp32/attendance", jsonString, 4000, &httpCode);
    if (httpCode == 200) {
      JsonDocument resDoc;
      deserializeJson(resDoc, responsePayload);
      bool ok = resDoc["success"].as<bool>();
      String msg = resDoc[ok ? "name" : "message"].as<String>();
      if (ok) {
        triggerIndicatorSafe(true, "Welcome!\n" + msg);
        logDataToSD(jsonString);
        if (resolvedStudentId != -1)
          addToOfflineCache(resolvedStudentId);
      } else {
        if (msg == "Already Scanned Today") {
          playAudio("duplicate");
          triggerIndicatorSafe(false, "Already Present\nToday!");
        } else {
          triggerIndicatorSafe(false, msg);
        }
      }
    } else {
      goto offline_cache;
    }
    return;
  }

  // Offline mode
  {
    if (resolvedStudentId == -1) {
      playAudio("error");
      String label = (sensorType == "fingerprint") ? "Finger" : "Tag";
      triggerIndicatorSafe(false, label + " Not Found\nIn Whitelist");
      return;
    }
    if (isDuplicateOfflineScan(resolvedStudentId)) {
      playAudio("duplicate");
      triggerIndicatorSafe(false, "Already Present\n(Offline)");
      return;
    }
  }

offline_cache: {
  if (resolvedStudentId != -1)
    addToOfflineCache(resolvedStudentId);
  JsonDocument doc;
  doc["sensor"] = sensorType;
  if (sensorType == "fingerprint")
    doc["id"] = identifier.toInt();
  else
    doc["uid"] = identifier;
  doc["timestamp"] = timestamp;
  String jsonString;
  serializeJson(doc, jsonString);
  logDataToSD(jsonString);
  if (xSemaphoreTake(sdMutex, portMAX_DELAY)) {
    File file = SD.open("/unsent.txt", FILE_APPEND);
    if (file) {
      file.println(jsonString);
      file.close();
    }
    xSemaphoreGive(sdMutex);
  }
  countUnsentRecords();
  triggerIndicatorSafe(true, "Saved Offline\nWill Sync Soon");
}
}

void logDataToSD(String jsonString) {
  if (!sdReady)
    return;
  if (xSemaphoreTake(sdMutex, portMAX_DELAY)) {
    File file = SD.open("/master_log.txt", FILE_APPEND);
    if (file) {
      file.println(jsonString);
      file.close();
    }
    xSemaphoreGive(sdMutex);
  }
}

void countUnsentRecords() {
  if (!sdReady)
    return;
  int count = 0;
  File file = SD.open("/unsent.txt", FILE_READ);
  if (file) {
    while (file.available()) {
      if (file.read() == '\n')
        count++;
    }
    file.close();
  }
  unsentCount = count;
}

// ========================= handleAttendance ========================
// Modified flow:
// RFID detected → TinyML check → if ok → prompt voice (9 s) → upload → server
// decides proxy
//              → if proxy → fingerprint audit
//              → if ok    → processAttendanceData
void handleAttendance() {
  // ── Security Audit Active ──
  if (pendingAudit) {
    int studentIndex = getWhitelistIndexByRFID(auditExpectedRFID);
    int expectedFinger =
        (studentIndex >= 0) ? whitelist[studentIndex].fingerprint_id : -1;

    // Timeout → blacklist
    if (millis() - auditStartTime > 15000) {
      pendingAudit = false;
      noTone(BUZZER);
      if (studentIndex >= 0)
        whitelist[studentIndex].is_blacklisted = true;

      JsonDocument doc;
      doc["uid"] = auditExpectedRFID;
      doc["ai_dwell"] = lastDwellTime;
      doc["ai_inter"] = lastInterScanTime;
      doc["ai_prediction"] = "PROXY";
      String jsonString;
      serializeJson(doc, jsonString);
      if (isOnline && serverBaseURL != "") {
        doHttpPost("/api/esp32/proxy_alert", jsonString, 4000);
        // Mark voice match as blacklisted
        JsonDocument resolveDoc;
        resolveDoc["rfid_uid"] = auditExpectedRFID;
        resolveDoc["result"] = "BLACKLISTED";
        String resolveJson;
        serializeJson(resolveDoc, resolveJson);
        doHttpPost("/api/esp32/voice_match_resolve", resolveJson, 3000);
      } else {
        doc["sensor"] = "proxy_alert";
        serializeJson(doc, jsonString);
        logDataToSD(jsonString);
      }
      triggerIndicatorSafe(false, "AUDIT FAILED\nBlacklisted!");
      return;
    }

    // Check Master Card
    uint8_t u[7];
    uint8_t uLen;
    if (nfc.readPassiveTargetID(PN532_MIFARE_ISO14443A, u, &uLen, 50)) {
      String uStr = "";
      for (uint8_t i = 0; i < uLen; i++) {
        if (u[i] < 0x10)
          uStr += "0";
        uStr += String(u[i], HEX);
      }
      uStr.toUpperCase();
      if (cachedMasterRFID != "" && uStr == cachedMasterRFID) {
        pendingAudit = false;
        noTone(BUZZER);
        if (auditExpectedRFID != "SIMULATED") {
          // Mark resolved
          JsonDocument resolveDoc;
          resolveDoc["rfid_uid"] = auditExpectedRFID;
          resolveDoc["result"] = "FINGERPRINT_OK";
          String resolveJson;
          serializeJson(resolveDoc, resolveJson);
          if (isOnline && serverBaseURL != "")
            doHttpPost("/api/esp32/voice_match_resolve", resolveJson, 3000);
          processAttendanceData("rfid", auditExpectedRFID);
        } else if (isOnline && serverBaseURL != "") {
          JsonDocument doc;
          doc["uid"] = "SIMULATED";
          doc["method"] = "OVERRIDE";
          String js;
          serializeJson(doc, js);
          doHttpPost("/api/esp32/audit_cleared", js, 2000);
        }
        triggerIndicatorSafe(true, "AUDIT CLEARED\nMaster Card OK");
        return;
      } else {
        triggerIndicatorSafe(false, "INVALID RFID\nUse Master Card");
        delay(800);
      }
    }

    // Check fingerprint
    if (finger.getImage() == FINGERPRINT_OK &&
        finger.image2Tz(1) == FINGERPRINT_OK) {
      if (finger.fingerSearch() == FINGERPRINT_OK) {
        bool isMaster = (finger.fingerID == masterFingerSlot);
        bool isStudent =
            (expectedFinger > 0 && finger.fingerID == expectedFinger);
        if (isMaster || isStudent) {
          pendingAudit = false;
          noTone(BUZZER);
          String clearMsg = isMaster ? "AUDIT CLEARED\nMaster Finger"
                                     : "AUDIT CLEARED\nIdentity OK";
          // Resolve voice match
          if (auditExpectedRFID != "SIMULATED") {
            JsonDocument resolveDoc;
            resolveDoc["rfid_uid"] = auditExpectedRFID;
            resolveDoc["result"] = "FINGERPRINT_OK";
            String resolveJson;
            serializeJson(resolveDoc, resolveJson);
            if (isOnline && serverBaseURL != "")
              doHttpPost("/api/esp32/voice_match_resolve", resolveJson, 3000);
            if (isStudent)
              processAttendanceData("rfid", auditExpectedRFID);
          } else if (isOnline && serverBaseURL != "") {
            JsonDocument doc;
            doc["uid"] = "SIMULATED";
            doc["method"] = "OVERRIDE";
            String js;
            serializeJson(doc, js);
            doHttpPost("/api/esp32/audit_cleared", js, 2000);
          }
          triggerIndicatorSafe(true, clearMsg);
        } else {
          // Wrong finger
          pendingAudit = false;
          noTone(BUZZER);
          if (studentIndex >= 0)
            whitelist[studentIndex].is_blacklisted = true;
          JsonDocument doc;
          doc["uid"] = auditExpectedRFID;
          doc["ai_dwell"] = lastDwellTime;
          doc["ai_inter"] = lastInterScanTime;
          doc["ai_prediction"] = "PROXY";
          String js;
          serializeJson(doc, js);
          if (isOnline && serverBaseURL != "") {
            doHttpPost("/api/esp32/proxy_alert", js, 4000);
            JsonDocument resolveDoc;
            resolveDoc["rfid_uid"] = auditExpectedRFID;
            resolveDoc["result"] = "BLACKLISTED";
            String resolveJson;
            serializeJson(resolveDoc, resolveJson);
            doHttpPost("/api/esp32/voice_match_resolve", resolveJson, 3000);
          } else {
            doc["sensor"] = "proxy_alert";
            serializeJson(doc, js);
            logDataToSD(js);
          }
          triggerIndicatorSafe(false, "WRONG FINGER\nBlacklisted!");
        }
      } else {
        triggerIndicatorSafe(false, "Unknown Finger\nTry Again");
      }
    }
    return;
  }

  // ── Fingerprint Primary Check ──
  uint8_t p = finger.getImage();
  if (p != FINGERPRINT_OK && p != FINGERPRINT_NOFINGER) {
      Serial.printf("[Fingerprint] getImage error: 0x%02X\n", p);
  }
  if (p == FINGERPRINT_OK) {
    uint8_t tz = finger.image2Tz();
    if (tz != FINGERPRINT_OK) {
        Serial.printf("[Fingerprint] image2Tz error: 0x%02X\n", tz);
    }
    if (tz == FINGERPRINT_OK) {
    if (finger.fingerSearch() == FINGERPRINT_OK) {
      processAttendanceData("fingerprint", String(finger.fingerID));
    } else {
      triggerIndicatorSafe(false, "Unknown Finger\nTry Again");
    }
    }
  }

  // ── RFID Fallback Check ──
  uint8_t uid[7] = {0};
  uint8_t uidLength;
  unsigned long scanStartTime = millis();

  if (nfc.readPassiveTargetID(PN532_MIFARE_ISO14443A, uid, &uidLength, 50)) {
    // Measure dwell time
    unsigned long dwellTime = 0;
    while (
        nfc.readPassiveTargetID(PN532_MIFARE_ISO14443A, uid, &uidLength, 50)) {
      dwellTime = millis() - scanStartTime;
    }
    unsigned long interScanTime = millis() - lastRFIDScanTime;
    lastRFIDScanTime = millis();

    String uidString = "";
    for (uint8_t i = 0; i < uidLength; i++) {
      if (uid[i] < 0x10)
        uidString += "0";
      uidString += String(uid[i], HEX);
    }
    uidString.toUpperCase();
    lastDwellTime = (float)dwellTime;
    lastInterScanTime = (float)interScanTime;

    // ── Offline Mode Gate ─────────────────────────────────────────────
    // In offline mode, voice proxy detection cannot run (server needed).
    // RFID attendance is therefore BLOCKED to prevent unverified proxies.
    // EXCEPTION: Master card always works offline for portal toggle.
    // Fingerprint attendance is handled at the bottom of handleAttendance().
    if (!isOnline) {
      if (cachedMasterRFID != "" && uidString == cachedMasterRFID) {
        // Master card: allow portal toggle even offline
        processAttendanceData("rfid", uidString);
      } else {
        // Student RFID: reject — proxy check not possible without server
        playAudio("error");
        triggerIndicatorSafe(false, "Offline Mode:\nFingerprint Only");
      }
      return;
    }
    // ─────────────────────────────────────────────────────────────────

    // TinyML behavioural proxy check (dwell/inter-scan)
    int isProxy = ProxyModel::predict((float)dwellTime, (float)interScanTime);
    if (isProxy == 1) {
      // Behavioural proxy — trigger audit immediately (no voice needed)
      pendingAudit = true;
      auditExpectedRFID = uidString;
      auditStartTime = millis();
      playAudio("error");
      int studentIndex = getWhitelistIndexByRFID(uidString);
      String sName = (studentIndex >= 0) ? String(whitelist[studentIndex].name)
                                         : "Unknown";
      if (xSemaphoreTake(displayMutex, portMAX_DELAY)) {
        display.clearDisplay();
        display.fillRect(0, 0, 128, 16, SSD1306_WHITE);
        display.setTextColor(SSD1306_BLACK, SSD1306_WHITE);
        display.setCursor(4, 4);
        display.print("SECURITY AUDIT");
        display.setTextColor(SSD1306_WHITE, SSD1306_BLACK);
        display.setCursor(0, 24);
        display.print("Proxy Suspected!");
        display.setCursor(0, 36);
        display.print(sName);
        display.setCursor(0, 48);
        display.print("Validate Finger!");
        display.display();
        xSemaphoreGive(displayMutex);
      }
      return;
    }

    // ── Master Card: skip voice check ──
    // If the scanned card is the master card, go straight to portal toggle.
    // No voice verification needed for admin.
    if (cachedMasterRFID != "" && uidString == cachedMasterRFID) {
      processAttendanceData("rfid", uidString);
      return;
    }

    int voiceAttempts = 0; // Max 1 retry

  voice_retry:
    // ── Voice Recording Step ──
    // Show prompt on OLED and kick off recording on Core 0
    if (xSemaphoreTake(displayMutex, portMAX_DELAY)) {
      display.clearDisplay();
      display.fillRect(0, 0, 128, 16, SSD1306_WHITE);
      display.setTextColor(SSD1306_BLACK, SSD1306_WHITE);
      display.setCursor(4, 4);
      display.print("  VOICE CHECK  ");
      display.setTextColor(SSD1306_WHITE, SSD1306_BLACK);
      display.setCursor(0, 22);
      display.print("Say: Present Sir");
      display.setCursor(0, 34);
      display.print("Recording 5 seconds");
      display.setCursor(14, 48);
      display.print("  [===      ]");
      display.display();
      xSemaphoreGive(displayMutex);
    }

    // Double-beep to signal recording start
    playAudio("recording");

    // Copy RFID to shared buffer for voiceTask context
    strncpy((char *)voiceRfidUid, uidString.c_str(), sizeof(voiceRfidUid) - 1);

    // Signal voiceTask (Core 0) to start recording
    xEventGroupClearBits(voiceEventGroup, VOICE_DONE_BIT | VOICE_ABORT_BIT);
    xEventGroupSetBits(voiceEventGroup, VOICE_REQUEST_BIT);

    // Animate countdown while Core 0 records (Core 1 stays responsive)
    unsigned long recordStart = millis();
    int lastSec = 0;
    while (true) {
      // Check if recording completed
      EventBits_t bits = xEventGroupGetBits(voiceEventGroup);
      if (bits & VOICE_DONE_BIT) {
        xEventGroupClearBits(voiceEventGroup, VOICE_DONE_BIT);
        break;
      }
      // Update progress bar on OLED
      int elapsed = (int)((millis() - recordStart) / 1000);
      if (elapsed != lastSec) {
        lastSec = elapsed;
        int bars = (elapsed * 10) / VOICE_RECORD_SECS;
        bars = bars > 10 ? 10 : bars;
        String bar = "[";
        for (int b = 0; b < bars; b++)
          bar += "=";
        for (int b = bars; b < 10; b++)
          bar += " ";
        bar += "]";
        if (xSemaphoreTake(displayMutex, pdMS_TO_TICKS(30))) {
          display.fillRect(0, 44, 128, 20, SSD1306_BLACK);
          display.setCursor(0, 48);
          char cBuf[24];
          snprintf(cBuf, sizeof(cBuf), "%ds %s", elapsed, bar.c_str());
          display.print(cBuf);
          display.display();
          xSemaphoreGive(displayMutex);
        }
      }
      // Safety timeout: abort if recording takes > 12 s
      if (millis() - recordStart > 12000) {
        xEventGroupSetBits(voiceEventGroup, VOICE_ABORT_BIT);
        // Wait for voiceTask to acknowledge abort
        xEventGroupWaitBits(voiceEventGroup, VOICE_DONE_BIT, pdTRUE, pdTRUE,
                            pdMS_TO_TICKS(2000));
        break;
      }
      vTaskDelay(50 / portTICK_PERIOD_MS);
    }

    // ── Voice Upload & Server Decision ──
    bool voiceProxy = false;
    bool voiceWrongPhrase = false;
    String voiceMatchedName = "";
    float voiceScore = 0.0f;

    if (voiceFileSaved && isOnline && serverBaseURL != "") {
      if (xSemaphoreTake(displayMutex, portMAX_DELAY)) {
        display.clearDisplay();
        display.fillRect(0, 0, 128, 16, SSD1306_WHITE);
        display.setTextColor(SSD1306_BLACK, SSD1306_WHITE);
        display.setCursor(8, 4);
        display.print(" Analyzing Voice ");
        display.setTextColor(SSD1306_WHITE, SSD1306_BLACK);
        display.setCursor(0, 26);
        display.print("Uploading & comparing");
        display.setCursor(0, 40);
        display.print("Please wait...");
        display.display();
        xSemaphoreGive(displayMutex);
      }

      uploadVoiceAndGetDecision(uidString.c_str(), voiceProxy, voiceWrongPhrase,
                                voiceMatchedName, voiceScore);
    }

    if (voiceWrongPhrase) {
      voiceAttempts++;
      playAudio("error");
      if (voiceAttempts >= 2) {
        // 2nd failure — stop asking, prompt fingerprint instead
        if (xSemaphoreTake(displayMutex, portMAX_DELAY)) {
          display.clearDisplay();
          display.fillRect(0, 0, 128, 16, SSD1306_WHITE);
          display.setTextColor(SSD1306_BLACK, SSD1306_WHITE);
          display.setCursor(8, 4);
          display.print("  VOICE FAILED  ");
          display.setTextColor(SSD1306_WHITE, SSD1306_BLACK);
          display.setCursor(0, 22);
          display.print("Please use your");
          display.setCursor(0, 34);
          display.print(" FINGERPRINT now");
          display.setCursor(0, 48);
          display.print("Place finger...");
          display.display();
          xSemaphoreGive(displayMutex);
        }
        // Wait 4 s for fingerprint scan
        delay(500);
        unsigned long fpWait = millis();
        while (millis() - fpWait < 4000) {
          if (finger.getImage() == FINGERPRINT_OK &&
              finger.image2Tz() == FINGERPRINT_OK) {
            if (finger.fingerSearch() == FINGERPRINT_OK) {
              processAttendanceData("fingerprint", String(finger.fingerID));
            } else {
              triggerIndicatorSafe(false, "Unknown Finger\nTry Again");
            }
            return;
          }
          delay(50);
        }
        triggerIndicatorSafe(false, "Voice+Finger\nFailed. Retry.");
        return;
      }
      // First wrong attempt — show warning and retry once more
      if (xSemaphoreTake(displayMutex, portMAX_DELAY)) {
        display.clearDisplay();
        display.fillRect(0, 0, 128, 16, SSD1306_WHITE);
        display.setTextColor(SSD1306_BLACK, SSD1306_WHITE);
        display.setCursor(8, 4);
        display.print("  WRONG WORD!  ");
        display.setTextColor(SSD1306_WHITE, SSD1306_BLACK);
        display.setCursor(0, 22);
        display.print("You did not say:");
        display.setCursor(0, 34);
        display.print(" 'Present Sir'");
        display.setCursor(0, 48);
        display.print("1 chance left!");
        display.display();
        xSemaphoreGive(displayMutex);
      }
      delay(2500);
      goto voice_retry;
    }

    if (voiceProxy) {
      // Voice-based proxy detected
      pendingAudit = true;
      auditExpectedRFID = uidString;
      auditStartTime = millis();
      playAudio("error");

      char scoreStr[12];
      snprintf(scoreStr, sizeof(scoreStr), "%.0f%%", voiceScore);

      if (xSemaphoreTake(displayMutex, portMAX_DELAY)) {
        display.clearDisplay();
        display.fillRect(0, 0, 128, 16, SSD1306_WHITE);
        display.setTextColor(SSD1306_BLACK, SSD1306_WHITE);
        display.setCursor(4, 4);
        display.print("VOICE PROXY ALERT");
        display.setTextColor(SSD1306_WHITE, SSD1306_BLACK);
        display.setCursor(0, 20);
        display.print("Match: ");
        display.print(voiceMatchedName.substring(0, 12));
        display.setCursor(0, 32);
        display.print("Score: ");
        display.print(scoreStr);
        display.setCursor(0, 44);
        display.print("Validate Fingerprint");
        display.display();
        xSemaphoreGive(displayMutex);
      }
      return;
    }

    // Voice OK (or offline/model unavailable) — proceed with normal attendance
    processAttendanceData("rfid", uidString);
    return;
  }
}

// ========================= Enrollment Handlers =====================
void handleCaptureRfid() {
  triggerIndicatorSafe(true, "Tap Master Card\non Scanner...");
  uint8_t uid[7];
  uint8_t uidLength;
  if (nfc.readPassiveTargetID(PN532_MIFARE_ISO14443A, uid, &uidLength, 5000)) {
    String uidString = "";
    for (uint8_t i = 0; i < uidLength; i++) {
      if (uid[i] < 0x10)
        uidString += "0";
      uidString += String(uid[i], HEX);
    }
    uidString.toUpperCase();
    JsonDocument doc;
    doc["sensor"] = "rfid";
    doc["uid"] = uidString;
    String jsonString;
    serializeJson(doc, jsonString);
    if (isOnline && serverBaseURL != "") {
      doHttpPost("/api/esp32/attendance", jsonString);
      triggerIndicatorSafe(true, "Tag Read\nSuccessfully!");
    }
  }
  currentMode = "attendance";
}

void handleRFIDEnrollment() {

  uint8_t uidLength;
  if (nfc.readPassiveTargetID(PN532_MIFARE_ISO14443A, uid, &uidLength, 8000)) {
    String uidString = "";
    for (uint8_t i = 0; i < uidLength; i++) {
      if (uid[i] < 0x10)
        uidString += "0";
      uidString += String(uid[i], HEX);
    }
    uidString.toUpperCase();
    JsonDocument doc;
    doc["sensor"] = "rfid";
    doc["uid"] = uidString;
    String jsonString;
    serializeJson(doc, jsonString);
    if (isOnline && serverBaseURL != "") {
      int httpCode = 0;
      String responsePayload =
          doHttpPost("/api/esp32/enroll_success", jsonString, 4000, &httpCode);
      if (httpCode == 200) {

        deserializeJson(res, responsePayload);

          triggerIndicatorSafe(true, "Tag Linked!");

      }
    }

    triggerIndicatorSafe(false, "Timeout!\nNo Tag Found");
  }
  currentMode = "attendance";
}

void handleMasterFingerEnrollment() {
  finger.getTemplateCount();
  int slot = finger.templateCount + 1;
  if (slot > 127)
    slot = 127;

  triggerIndicatorSafe(true, "Demo Finger\nEnrollment...");
  vTaskDelay(1000 / portTICK_PERIOD_MS);

  triggerIndicatorSafe(true, "Place Finger\nOn Sensor (1/2)");
  int p = -1;
  while (p != FINGERPRINT_OK) {
    if (currentMode != "enroll_master_finger")
      return;
    p = finger.getImage();
    if (p != FINGERPRINT_OK && p != FINGERPRINT_NOFINGER) {
      Serial.printf("[Fingerprint Enroll] error: 0x%02X\n", p);
    }
    vTaskDelay(50 / portTICK_PERIOD_MS);
  }
  if (finger.image2Tz(1) != FINGERPRINT_OK) {
    triggerIndicatorSafe(false, "Scan failed\nTry Again");
    currentMode = "attendance";
    return;
  }

  playAudio("tech_init");
  triggerIndicatorSafe(true, "Good! Now\nLIFT FINGER!");
  p = 0;
  while (p != FINGERPRINT_NOFINGER) {
    if (currentMode != "enroll_master_finger")
      return;
    p = finger.getImage();
    if (p != FINGERPRINT_OK && p != FINGERPRINT_NOFINGER) {
      Serial.printf("[Fingerprint Enroll] error: 0x%02X\n", p);
    }
    vTaskDelay(50 / portTICK_PERIOD_MS);
  }

  triggerIndicatorSafe(true, "Place SAME\nFinger Again (2/2)");
  p = -1;
  while (p != FINGERPRINT_OK) {
    if (currentMode != "enroll_master_finger")
      return;
    p = finger.getImage();
    if (p != FINGERPRINT_OK && p != FINGERPRINT_NOFINGER) {
      Serial.printf("[Fingerprint Enroll] error: 0x%02X\n", p);
    }
    vTaskDelay(50 / portTICK_PERIOD_MS);
  }
  if (finger.image2Tz(2) != FINGERPRINT_OK) {
    triggerIndicatorSafe(false, "Scan failed\nTry Again");
    currentMode = "attendance";
    return;
  }

  if (finger.createModel() != FINGERPRINT_OK) {
    triggerIndicatorSafe(false, "Mismatch!\nPress Harder");
    currentMode = "attendance";
    return;
  }
  if (finger.storeModel(slot) != FINGERPRINT_OK) {
    triggerIndicatorSafe(false, "Store Failed!\nSensor Full?");
    currentMode = "attendance";
    return;
  }

  preferences.putInt("mfSlot", slot);
  masterFingerSlot = slot;
  playAudio("success");
  triggerIndicatorSafe(true, "Demo Finger OK!\nSlot: " + String(slot));

  JsonDocument doc;
  doc["sensor"] = "fingerprint";
  doc["id"] = slot;
  String jsonString;
  serializeJson(doc, jsonString);
  if (isOnline && serverBaseURL != "")
    doHttpPost("/api/esp32/enroll_success", jsonString, 4000);

  vTaskDelay(2000 / portTICK_PERIOD_MS);
  currentMode = "attendance";
}

// =====================================================================
// Test Recording — triggered by dashboard Voice Testing Lab
// Records 5 seconds from INMP441 mic, uploads to /api/esp32/voice_test_submit
// =====================================================================
void handleTestRecord() {
  // Immediately switch back so the mode doesn't re-trigger on next loop
  currentMode = "attendance";

  if (!isOnline || serverBaseURL == "") {
    triggerIndicatorSafe(true, "Test Rec FAIL\nNo Server");
    return;
  }

  // Show recording indicator on OLED
  if (xSemaphoreTake(displayMutex, portMAX_DELAY)) {
    display.clearDisplay();
    display.setTextSize(1);
    display.setCursor(0, 0);
    display.println("[TEST RECORD]");
    display.println("Recording 9s...");
    display.println("ESP32 Mic");
    display.display();
    xSemaphoreGive(displayMutex);
  }
  tone(BUZZER, 880, 150);

  // Trigger voice recording task (reuse existing mechanism)
  xEventGroupClearBits(voiceEventGroup, VOICE_DONE_BIT | VOICE_ABORT_BIT);
  xEventGroupSetBits(voiceEventGroup, VOICE_REQUEST_BIT);

  // Wait for recording to finish (up to 12 s)
  unsigned long waitStart = millis();
  while (millis() - waitStart < 12000) {
    EventBits_t bits = xEventGroupGetBits(voiceEventGroup);
    if (bits & VOICE_DONE_BIT) {
      xEventGroupClearBits(voiceEventGroup, VOICE_DONE_BIT);
      break;
    }
    vTaskDelay(100 / portTICK_PERIOD_MS);
  }

  if (!voiceFileSaved) {
    triggerIndicatorSafe(true, "Test Rec FAIL\nNo Audio");
    return;
  }

  // Give SD SPI bus 200ms to fully settle after the recording write
  // before doHttpPostMultipart tries to open and stream the file
  vTaskDelay(200 / portTICK_PERIOD_MS);

  if (xSemaphoreTake(displayMutex, portMAX_DELAY)) {
    display.clearDisplay();
    display.setCursor(0, 0);
    display.println("[TEST RECORD]");
    display.println("Uploading...");
    display.display();
    xSemaphoreGive(displayMutex);
  }

  // Reuse doHttpPostMultipart but target /api/esp32/voice_test_submit
  String responseOut;
  bool ok = doHttpPostMultipart("/api/esp32/voice_test_submit", "test", 30000, responseOut);

  if (ok) {
    tone(BUZZER, 1200, 200);
    delay(100);
    tone(BUZZER, 1500, 150);
    triggerIndicatorSafe(true, "Test Rec OK!\nUploaded");
  } else {
    tone(BUZZER, 300, 400);
    triggerIndicatorSafe(true, "Test Rec FAIL\nUpload Err");
  }
}
